// 1. 날짜 보정 및 형식화 함수 (타임존 오차 완벽 해결) function normDate(v) { if (!v) return null; // Date 객체인 경우 (엑셀에서 읽어온 날짜) if (v instanceof Date) { // 엑셀의 UTC 자정 데이터를 한국 시간(KST)으로 변환하기 위해 9시간 추가 const kstOffset = 9 * 60 * 60 * 1000; const kstDate = new Date(v.getTime() + kstOffset); // ISO 형식(YYYY-MM-DD)으로 변환 const y = kstDate.getUTCFullYear(); const m = String(kstDate.getUTCMonth() + 1).padStart(2, '0'); const d = String(kstDate.getUTCDate()).padStart(2, '0'); return `${y}-${m}-${d}`; } // 문자열인 경우 (예: "2026년 3월 31일" 또는 "2026-03-31") const s = String(v).trim(); const m = s.match(/(\d{4})[년\s./-]*(\d{1,2})[월\s./-]*(\d{1,2})?/); if (m) { const year = m[1]; const month = m[2].padStart(2, '0'); const day = m[3] ? m[3].padStart(2, '0') : "01"; // 일자 없으면 1일로 처리 return `${year}-${month}-${day}`; } return null; } // 2. 엑셀 메타데이터(날짜, 서비스월) 추출 로직 (이미지의 AE, AD열 대응) function getExcelMeta(wb) { let meta = { 작성일자: null, 서비스월: null, 정산월YYYYMM: null }; // 첫 번째 시트를 기준으로 데이터를 정밀 분석 const sheetName = wb.SheetNames[0]; const ws = wb.Sheets[sheetName]; // header:1 옵션으로 모든 셀을 배열로 가져옴 const data = XLSX.utils.sheet_to_json(ws, { header: 1, defval: null, UTC: true }); // 데이터 상단 20줄을 스캔하여 '정산월', '서비스월' 키워드와 날짜 데이터 매칭 for (let r = 0; r < Math.min(data.length, 20); r++) { const row = data[r]; if (!row) continue; for (let c = 0; r < row.length; c++) { const val = String(row[c] || "").trim(); // 작성일자(정산월) 찾기: 이미지의 AE열(index 30) 근처 혹은 키워드 매칭 if (val.includes("정산월") || c === 30) { // 현재 셀이 날짜면 그대로 사용, 아니면 바로 아래 셀 확인 const dateVal = (val.match(/\d{4}/)) ? row[c] : (data[r + 1] ? data[r + 1][c] : null); if (dateVal && !meta.작성일자) meta.작성일자 = normDate(dateVal); } // 서비스월 찾기: 이미지의 AD열(index 29) 근처 혹은 키워드 매칭 if (val.includes("서비스월") || c === 29) { const srvVal = (val.match(/\d{4}/)) ? row[c] : (data[r + 1] ? data[r + 1][c] : null); if (srvVal && !meta.서비스월) meta.서비스월 = normDate(srvVal); } } if (meta.작성일자 && meta.서비스월) break; } // 최종 결과에서 YYYYMM 추출 if (meta.작성일자) meta.정산월YYYYMM = meta.작성일자.substring(0, 7).replace('-', ''); return meta; }
